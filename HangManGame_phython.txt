import random

words = {
    "Programming": ["python", "java", "javascript", "compiler", "algorithm"],
    "Animals": ["elephant", "giraffe", "kangaroo", "dolphin", "penguin"],
    "Countries": ["india", "canada", "australia", "brazil", "germany"]
}

wins = 0
losses = 0

def play_hangman():
    global wins, losses

    category = random.choice(list(words.keys()))
    word = random.choice(words[category])

    guessed_letters = set()
    correct_letters = set(word)
    lives = 6

    print(f"\nCategory: {category}")
    print("Welcome to Hangman!")

    while lives > 0:
        display_word = " ".join(
            letter if letter in guessed_letters else "_"
            for letter in word
        )

        print("\nWord:", display_word)
        print("Guessed Letters:", " ".join(sorted(guessed_letters)))
        print("Lives Remaining:", lives)

        if correct_letters.issubset(guessed_letters):
            print(f"\n🎉 Congratulations! You guessed the word '{word}'.")
            wins += 1
            return

        guess = input("Enter a letter: ").lower().strip()

        if len(guess) != 1 or not guess.isalpha():
            print("Please enter a single alphabet letter.")
            continue

        if guess in guessed_letters:
            print("You already guessed that letter.")
            continue

        guessed_letters.add(guess)

        if guess not in word:
            lives -= 1
            print("❌ Wrong guess!")

    print(f"\n💀 Game Over! The word was '{word}'.")
    losses += 1

while True:
    play_hangman()

    print(f"\nScoreboard -> Wins: {wins} | Losses: {losses}")

    choice = input("\nDo you want to play again? (yes/no): ").lower()

    if choice != "yes":
        print("\nThanks for playing Hangman!")
        break
